interface SearchEngine {
  id: string;
  name: string;
  icon: React.ReactNode;
  url: string;
  placeholder: string;
}

const searchEngines: SearchEngine[] = [
  {
    id: 'google',
    name: 'Google',
    icon: <GoogleIcon />,
    url: 'https://www.google.com/search?q=',
    placeholder: 'åœ¨Googleä¸­æœç´¢...'
  },
  // ... å…¶ä»–æœç´¢å¼•æ“
];

const SearchBox: React.FC = () => {
  const [searchQuery, setSearchQuery] = useState('');
  const [selectedEngine, setSelectedEngine] = useState('google');
  const [showDropdown, setShowDropdown] = useState(false);
  const [hoveredEngine, setHoveredEngine] = useState<string | null>(null);
  const containerRef = useRef<HTMLDivElement>(null);

  const currentEngine = searchEngines.find(e => e.id === selectedEngine) || searchEngines[0];

  // å¤„ç†æœç´¢æäº¤
  const handleSearchSubmit = () => {
    if (searchQuery.trim()) {
      const searchUrl = currentEngine.url + encodeURIComponent(searchQuery.trim());
      window.open(searchUrl, '_blank');
    }
  };

  // å¤„ç† Enter é”®
  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter') {
      handleSearchSubmit();
    }
  };

  // é€‰æ‹©æœç´¢å¼•æ“
  const selectEngine = (engine: SearchEngine) => {
    setSelectedEngine(engine.id);
    setShowDropdown(false);
  };

  // ç‚¹å‡»å¤–éƒ¨å…³é—­ä¸‹æ‹‰èœå•
  useEffect(() => {
    const handleClickOutside = (event: MouseEvent) => {
      if (containerRef.current && !containerRef.current.contains(event.target as Node)) {
        setShowDropdown(false);
      }
    };

    document.addEventListener('mousedown', handleClickOutside);
    return () => document.removeEventListener('mousedown', handleClickOutside);
  }, []);

  return (
    <div className="search-section">
      <div className="search-section-container">
        <div className="search-container" ref={containerRef}>
          {/* æœç´¢å¼•æ“é€‰æ‹©å™¨ */}
          <div className="search-engine-selector-left">
            <div 
              className="current-engine" 
              onClick={() => setShowDropdown(!showDropdown)}
            >
              {currentEngine.icon}
            </div>
          </div>

          {/* æœç´¢æ¡†ä¸»ä½“ */}
          <div className="search-main-area">
            <input
              type="text"
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
              onKeyDown={handleKeyDown}
              placeholder={showDropdown ? '' : currentEngine.placeholder}
              className={`search-input ${showDropdown ? 'hide-text' : ''}`}
            />

            {/* æ¸…é™¤æŒ‰é’® */}
            {searchQuery && !showDropdown && (
              <button
                onClick={() => setSearchQuery('')}
                className="clear-button"
                title="æ¸…é™¤"
              >
                <svg xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
                  <line x1="18" y1="6" x2="6" y2="18"></line>
                  <line x1="6" y1="6" x2="18" y2="18"></line>
                </svg>
              </button>
            )}

            {/* å†…éƒ¨ä¸‹æ‹‰èœå• */}
            {showDropdown && (
              <div className={`internal-dropdown dropdown-enter`}>
                <div className="engines-grid">
                  {searchEngines
                    .filter(e => e.id !== selectedEngine)
                    .map((engine, index) => (
                      <div
                        key={engine.id}
                        className="engine-option"
                        style={{ animationDelay: `${index * 0.05}s` }}

                      </div>
                    ))}
                </div>

                {/* æ‚¬åœæç¤º */}
                {hoveredEngine && (
                  <div className="engine-tooltip">{hoveredEngine}</div>
                )}
              </div>
            )}
          </div>

          {/* æœç´¢æŒ‰é’® */}
          <button 
            onClick={handleSearchSubmit}
            className="search-button"
            type="button"
          >
            <i className="fa-solid fa-magnifying-glass"></i>
          </button>
        </div>
      </div>
    </div>
  );
